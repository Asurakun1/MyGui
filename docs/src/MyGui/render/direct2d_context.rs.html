15/// Manages all Direct2D and DirectWrite resources.
16///
17/// This struct encapsulates the factories, render targets, and other resources
18/// required for drawing. It separates resource creation into two categories:
19/// - **Device-independent resources**: These resources (like `IDWriteTextFormat`)
20///   do not depend on the specific rendering device and can be created once.
21/// - **Device-dependent resources**: These resources (like `ID2D1HwndRenderTarget`
22///   and brushes) are tied to a specific display device. They may need to be

24/// Manages all Direct2D and DirectWrite resources.
25///
26/// This struct encapsulates the factories, render targets, and other resources
27/// required for drawing. It separates resource creation into two categories:
28/// - **Device-independent resources**: These resources (like `IDWriteTextFormat`)
29///   do not depend on the specific rendering device and can be created once.
30/// - **Device-dependent resources**: These resources (like `ID2D1HwndRenderTarget`
31///   and brushes) are tied to a specific display device. They may need to be
32///   recreated if the device is lost.
33pub struct Direct2DContext {
34    // Device-independent resources
35    pub d2d_factory: ID2D1Factory1,
36    pub dwrite_factory: IDWriteFactory,
37    pub text_format: Option<IDWriteTextFormat>,
38
39    // Device-dependent resources
40    pub render_target: Option<ID2D1HwndRenderTarget>,
41    pub brush: Option<ID2D1SolidColorBrush>,
42}
43
44impl Direct2DContext {
45    /// Creates a new `Direct2DContext` and initializes device-independent resources.
46    pub fn new() -> Result<Self> {
47        unsafe {
48            CoInitializeEx(None, COINIT_APARTMENTTHREADED).ok()?;
49        }
50
51        let d2d_factory_options = D2D1_FACTORY_OPTIONS {
52            debugLevel: if cfg!(debug_assertions) {
53                D2D1_DEBUG_LEVEL_INFORMATION
54            } else {
55                D2D1_DEBUG_LEVEL_NONE
56            },
57        };
58
59        let d2d_factory: ID2D1Factory1 = unsafe {
60            D2D1CreateFactory(
61                D2D1_FACTORY_TYPE_SINGLE_THREADED,
62                Some(&d2d_factory_options),
63            )?
64        };
65
66        let dwrite_factory: IDWriteFactory = unsafe {
67            DWriteCreateFactory(DWRITE_FACTORY_TYPE_SHARED)?
68        };
69
70        let mut context = Self {
71            d2d_factory,
72            dwrite_factory,
73            render_target: None,
74            text_format: None,
75            brush: None,
76        };
77
78        context.create_device_independent_resources()?;
79
80        Ok(context)
81    }
82
83    /// Creates resources that are not tied to a specific rendering device.
84    /// These resources can be created once and reused.
85    fn create_device_independent_resources(&mut self) -> Result<()> {
86        // Create a DirectWrite text format object.
87        // Unsafe: FFI call to DirectWrite. Assumes `dwrite_factory` is a valid pointer.
88        let text_format = unsafe {
89            self.dwrite_factory.CreateTextFormat(
90                &HSTRING::from(FONT_FACE_NAME),
91                None,
92                DWRITE_FONT_WEIGHT_NORMAL,
93                DWRITE_FONT_STYLE_NORMAL,
94                DWRITE_FONT_STRETCH_NORMAL,
95                FONT_SIZE as f32,
96                &HSTRING::from("en-us"),
97            )?
98        };
99        self.text_format = Some(text_format);
100        Ok(())
101    }
102
103    /// Creates resources that are tied to a specific rendering device (the `HWND`).
104    /// These resources need to be recreated if the device becomes invalid,
105    /// for example, when the window is resized.
106    pub fn create_device_dependent_resources(&mut self, hwnd: HWND) -> Result<()> {
107        // Get the size of the window's client area.
108        let mut rect = RECT::default();
109        // Unsafe: FFI call to GetClientRect. Assumes `hwnd` is a valid window handle.
110        unsafe { GetClientRect(hwnd, &mut rect)? };
111
112        let render_target_properties = D2D1_RENDER_TARGET_PROPERTIES::default();
113
114        let hwnd_render_target_properties = D2D1_HWND_RENDER_TARGET_PROPERTIES {
115            hwnd,
116            pixelSize: D2D_SIZE_U {
117                width: (rect.right - rect.left) as u32,
118                height: (rect.bottom - rect.top) as u32,
119            },
120            presentOptions: D2D1_PRESENT_OPTIONS_NONE,
121        };
122
123        // Create a Direct2D render target that can draw to the specified window.
124        // Unsafe: FFI call to Direct2D. Assumes `d2d_factory` is a valid pointer
125        // and the properties structs are correctly initialized.
126        let render_target = unsafe {
127            let factory = self.d2d_factory.cast::<ID2D1Factory>()?;
128            factory.CreateHwndRenderTarget(
129                &render_target_properties,
130                &hwnd_render_target_properties,
131            )?
132        };
133
134        // Create a solid color brush for drawing.
135        // Unsafe: FFI call to Direct2D. Assumes the render target is valid.
136        let brush = unsafe {
137            let rt: &ID2D1RenderTarget = &render_target;
138            rt.CreateSolidColorBrush(&D2D1_COLOR_F { r: 1.0, g: 1.0, b: 1.0, a: 1.0 }, None)?
139        };
140
141        self.render_target = Some(render_target);
142        self.brush = Some(brush);
143
144        Ok(())
145    }
146}
