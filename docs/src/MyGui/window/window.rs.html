16/// Represents an application window.
17///
18/// This struct encapsulates the window handle (`HWND`) and all the resources
19/// required to manage the window. It also owns the application state (`App`)
20/// and the event handler (`EventHandler`).
21pub struct Window<E: EventHandler> {
22    pub hwnd: HWND,
23    pub d2d_context: Direct2DContext,
24    pub event_handler: E,
25    pub app: App,
26}
27
28impl<E: EventHandler + 'static> Window<E> {
29    /// Creates a new application window.
30    ///
31    /// This function orchestrates the entire window setup process:
32    /// 1. Registers the window class with the operating system.
33    /// 2. Creates the `Window` struct on the heap using `Box::new`.
34    /// 3. Creates the actual Win32 window using `CreateWindowExW`.
35    ///    - A raw pointer to the heap-allocated `Window` struct is passed as the
36    ///      `lpParam` argument. This pointer is retrieved in `wndproc` during the
37    ///      `WM_NCCREATE` message to associate the Rust struct with the `HWND`.
38    /// 4. Initializes device-dependent Direct2D resources.
39    /// 5. Shows and updates the window.
40    ///
41    /// The returned `Box<Self>` is the sole owner of the `Window` struct at this point.
42    /// However, its lifetime will be managed by the `wndproc` and the message loop,
43    /// so the caller is expected to call `std::mem::forget` on the box to prevent
44    /// premature deallocation.
45    pub fn new(title: &str, class_name: &str, event_handler: E, app: App) -> Result<Box<Self>> {
46        let instance = unsafe { GetModuleHandleW(None)? };
47        Self::register_class(instance.into(), class_name)?;
48
49        // Allocate the Window struct on the heap. This is necessary because its lifetime
50        // will be tied to the Win32 window handle, not the scope of this function.
51        let mut window = Box::new(Self {
52            hwnd: HWND(std::ptr::null_mut()),
53            d2d_context: Direct2DContext::new()?,
54            event_handler,
55            app,
56        });
57
58        // The `CreateWindowExW` function will send a `WM_NCCREATE` message before it returns.
59        // Our `wndproc` handles this message by storing the pointer to our `window` box
60        // in the window's user data area (`GWLP_USERDATA`).
61        let hwnd = unsafe {
62            CreateWindowExW(
63                WINDOW_EX_STYLE::default(),
64                &HSTRING::from(class_name),
65                &HSTRING::from(title),
66                WS_OVERLAPPEDWINDOW,
67                CW_USEDEFAULT,
68                CW_USEDEFAULT,
69                WINDOW_WIDTH,
70                WINDOW_HEIGHT,
71                None,
72                None,
73                Some(instance.into()),
74                // Pass a pointer to the heap-allocated `Window` struct.
75                // This will be received by `wndproc` as the `lParam` of the `WM_NCCREATE` message.
76                Some(window.as_mut() as *mut _ as *mut _),
77            )
78        }?;
79
80        // Now that the Win32 window is created, store its handle in our struct.
81        window.hwnd = hwnd;
82        window.d2d_context.create_device_dependent_resources(hwnd)?;
83
84        unsafe {
85            let _ = ShowWindow(hwnd, SW_SHOW);
86        };
87        unsafe {
88            let _ = UpdateWindow(hwnd);
89        };
90
91        Ok(window)
92    }
93
94    fn register_class(instance: HINSTANCE, class_name: &str) -> Result<()> {
95        let class_name_hstring = HSTRING::from(class_name);
96
97        let wc = WNDCLASSEXW {
98            cbSize: std::mem::size_of::<WNDCLASSEXW>() as u32,
99            style: CS_HREDRAW | CS_VREDRAW,
100            lpfnWndProc: Some(wndproc::<E>),
101            cbClsExtra: 0,
102            // Allocate extra memory for the window instance. This is used to store a pointer
103            // to our heap-allocated `Window` struct. The pointer is set in `wndproc` during
104            // `WM_NCCREATE` and retrieved later to access the `Window`'s state.
105            cbWndExtra: std::mem::size_of::<*mut Self>() as i32,
106            hInstance: instance,
107            hIcon: unsafe { LoadIconW(None, IDI_APPLICATION)? },
108            hCursor: unsafe { LoadCursorW(None, IDC_ARROW)? },
109            hbrBackground: unsafe { HBRUSH(GetStockObject(BLACK_BRUSH).0) },
110            lpszMenuName: PCWSTR::null(),
111            lpszClassName: PCWSTR::from_raw(class_name_hstring.as_ptr()),
112            hIconSm: unsafe { LoadIconW(None, IDI_APPLICATION)? },
113        };
114
115        // Register the window class.
116        unsafe {
117            if RegisterClassExW(&wc) == 0 {
118                return Err(Error::from_hresult(HRESULT::from_win32(GetLastError().0)));
119            }
120        }
121
122        Ok(())
123    }
124
125    /// Starts the message loop for the window.
126    pub fn message_loop(&self) -> Result<()> {
127        let mut message = MSG::default();
128        // The main message loop.
129        while unsafe { GetMessageW(&mut message, None, 0, 0) }.into() {
130            unsafe { let _ = TranslateMessage(&message); };
131            unsafe { DispatchMessageW(&message) };
132        }
133        Ok(())
134    }
