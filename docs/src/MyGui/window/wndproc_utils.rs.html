11/// The main window procedure (`wndproc`) for the application.
12///
13/// This function is the central callback that receives all window messages from the OS.
14/// It's responsible for two key tasks:
15/// 1. Associating the `HWND` with the Rust `Window` struct instance. This is done
16///    during the `WM_NCCREATE` message by storing a pointer to the `Window` struct
17///    in the window's user data area (`GWLP_USERDATA`).
18/// 2. Dispatching messages to the `EventHandler` associated with the `Window`.
19///
20/// # Safety
21///
22/// This function contains significant `unsafe` code because it directly interacts
23/// with the Win32 API and manages raw pointers. The primary safety mechanism is
24/// the careful management of the `Window` pointer stored in `GWLP_USERDATA`.
25/// The pointer is set on creation and is valid until `WM_NCDESTROY`, at which point
26/// it is retrieved, converted back into a `Box`, and dropped by Rust, ensuring
27/// proper cleanup.
28pub extern "system" fn wndproc<E: EventHandler + 'static>(
29    hwnd: HWND,
30    message: u32,
31    wparam: WPARAM,
32    lparam: LPARAM,
33) -> LRESULT {
34    // Retrieve the pointer to the `Window` struct.
35    // On the first message (WM_NCCREATE), we get it from the CREATESTRUCT `lParam`
36    // and store it in the window's user data.
37    // For all subsequent messages, we retrieve it from the user data.
38    let window = unsafe {
39        if message == WM_NCCREATE {
40            let createstruct = lparam.0 as *const CREATESTRUCTW;
41            let window = (*createstruct).lpCreateParams as *mut Window<E>;
42            SetWindowLongPtrW(hwnd, GWLP_USERDATA, window as _);
43            window
44        } else {
45            GetWindowLongPtrW(hwnd, GWLP_USERDATA) as *mut Window<E>
46        }
47    };
48
49    // If the pointer is null, it means we are processing messages for a window
50    // before our state is fully initialized. In this case, we pass the message
51    // to the default window procedure.
52    if window.is_null() {
53        return unsafe { DefWindowProcW(hwnd, message, wparam, lparam) };
54    }
55
56    // Dereference the raw pointer to get a mutable reference to our `Window` struct.
57    // This is safe because the pointer is guaranteed to be valid from WM_NCCREATE
58    // until WM_NCDESTROY.
59    let window = unsafe { &mut *window };
60
61    match message {
62        // WM_PAINT is sent when the window needs to be redrawn.
63        WM_PAINT => {
64            if let (Some(render_target), Some(brush), Some(text_format)) = (
65                &window.d2d_context.render_target,
66                &window.d2d_context.brush,
67                &window.d2d_context.text_format,
68            ) {
69                let drawing_context = DrawingContext {
70                    render_target,
71                    brush,
72                    text_format,
73                };
74
75                window
76                    .event_handler
77                    .on_paint(&mut window.app, &drawing_context);
78            }
79            LRESULT(0)
80        }
81        // WM_SIZE is sent when the window has been resized.
82        WM_SIZE => {
83            let width = (lparam.0 & 0xFFFF) as i32;
84            let height = ((lparam.0 >> 16) & 0xFFFF) as i32;
85            window
86                .event_handler
87                .on_resize(&mut window.app, width, height);
88            // Resize the render target to match the new window size.
89            if let Some(render_target) = &window.d2d_context.render_target {
90                let new_size = D2D_SIZE_U {
91                    width: width as u32,
92                    height: height as u32,
93                };
94                unsafe { render_target.Resize(&new_size).ok() };
95            }
96            LRESULT(0)
97        }
98        // WM_MOUSEMOVE is sent when the mouse moves.
99        WM_MOUSEMOVE => {
100            let x = (lparam.0 & 0xFFFF) as i32;
101            let y = ((lparam.0 >> 16) & 0xFFFF) as i32;
102            window.event_handler.on_mouse_move(&mut window.app, x, y);
103            LRESULT(0)
104        }
105        // WM_LBUTTONDOWN is sent when the left mouse button is pressed.
106        WM_LBUTTONDOWN => {
107            let x = (lparam.0 & 0xFFFF) as i32;
108            let y = ((lparam.0 >> 16) & 0xFFFF) as i32;
109            window.event_handler.on_lbutton_down(&mut window.app, x, y);
110            LRESULT(0)
111        }
112        // WM_LBUTTONUP is sent when the left mouse button is released.
113        WM_LBUTTONUP => {
114            let x = (lparam.0 & 0xFFFF) as i32;
115            let y = ((lparam.0 >> 16) & 0xFFFF) as i32;
116            window.event_handler.on_lbutton_up(&mut window.app, x, y);
117            LRESULT(0)
118        }
119        // WM_KEYDOWN is sent when a key is pressed.
120        WM_KEYDOWN => {
121            let key = KeyId::from_vkey(wparam.0 as u16);
122            window.event_handler.on_key_down(&mut window.app, key);
123            LRESULT(0)
124        }
125        // WM_KEYUP is sent when a key is released.
126        WM_KEYUP => {
127            let key = KeyId::from_vkey(wparam.0 as u16);
128            window.event_handler.on_key_up(&mut window.app, key);
129            LRESULT(0)
130        }
131        // WM_DESTROY is sent when the window is being destroyed.
132        WM_DESTROY => {
133            window.event_handler.on_destroy(&mut window.app);
134            // Post a quit message to the message loop to terminate the application.
135            unsafe { PostQuitMessage(0) };
136            LRESULT(0)
137        }
138        // WM_NCDESTROY is the last message a window receives.
139        // This is the appropriate place to clean up our `Window` struct.
140        WM_NCDESTROY => {
141            // Retrieve the pointer from user data and clear it.
142            let ptr = unsafe { SetWindowLongPtrW(hwnd, GWLP_USERDATA, 0) };
143            if ptr != 0 {
144                // Re-constitute the Box from the raw pointer and let Rust drop it,
145                // freeing the memory. This is the crucial cleanup step.
146                let _ = unsafe { Box::from_raw(ptr as *mut Window<E>) };
147            }
148            LRESULT(0)
149        }
150        // For any other message, delegate to our event handler.
151        _ => {
152            if let Some(result) =
153                window
154                    .event_handler
155                    .handle_message(&mut window.app, message, wparam, lparam)
156            {
157                return LRESULT(result);
158            }
159            unsafe { DefWindowProcW(hwnd, message, wparam, lparam) }
160        }
161    }
162}
